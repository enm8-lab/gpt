(* Настройки AI блока
	ChangeRange - Диапазон: 0 - диапазон 4..20 мА (0..20 мА), 1 - диапазон 0..10 В, 2 - диапазон 1..5 В, 3 - диапазон 1..5 В (под диапазон 0..10 В);
	Hyst -  Гистерезис;
	TIME_HH	- Время задержки для формирования сигнализации HH;
	TIME_H - Время задержки для формирования сигнализации H;
	TIME_L - Время задержки для формирования сигнализации L;
	TIME_LL - Время задержки для формирования сигнализации LL;
	PERC_OF_DISCREP - Максимальное значние расхождения тока, %;
	ERROR_HANDLING - Что делать при возникновении неисправности: 0 - KEEP_LAST_VAL (Сохранять последнее значение); 1 - SET_TO_ZERO (Устанавливать значение низа шкалы); 2 - CONTINUE_EVAL (Продолжать вычисления);
	FILTER_CONSTANT - Конастанта фильтра НЧ (0; 1] (минимальное значение 0.01);
	IS_REDUND - Блок используется для резервирования;
	HIGH_DEBLOCK - Деблокировать верхние сигнализации H, HH ;
	LOW_DEBLOCK - Деблокировать нижние сигнализации L, LL ;
*)
TYPE AI_SETTINGS:
STRUCT
	Hyst			: REAL := 2;		// Гистерезис, % от шкалы
	TIME_HH 		: TIME := T#1S;
	TIME_H 			: TIME := T#1S;
	TIME_L 			: TIME := T#1S;
	TIME_LL 		: TIME := T#1S;
	PERC_OF_DISCREP : REAL := 2.0;		// Максимальное значние расхождения тока, %
	ERROR_HANDLING 	: AI_ERR_HAND; // Что делать при возникновении неисправности
	FILTER_CONSTANT : REAL := 0.5; // Конастанта фильтра НЧ (0; 1] (минимальное значение 0.01);	
	HIGH_DEBLOCK	: BOOL; // Деблокировка верхних порогов HH, H - 1 включить / 0 отключить;
	LOW_DEBLOCK		: BOOL; // Деблокировка нижних порогов LL, L - 1 включить / 0 отключить;
	ChangeRange		: BYTE;	// 0 - диапазон 4..20 мА (0..20 мА), 1 - диапазон 0..10 В, 2 - диапазон 1..5 В, 3 - диапазон 1..5 В (под диапазон 0..10 В)	
	IS_REDUND		: BOOL; // Блок используется для резервирования
END_STRUCT
END_TYPE
